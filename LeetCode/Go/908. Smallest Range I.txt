func smallestRangeI(nums []int, k int) int {

	min, max := MinMax(nums)
	return int(math.Max(float64(max-min-2*k), 0))
}

func MinMax(array []int) (int, int) {
	var max int = array[0]
	var min int = array[0]
	for _, value := range array {
		if max < value {
			max = value
		}
		if min > value {
			min = value
		}
	}
	return min, max
}
