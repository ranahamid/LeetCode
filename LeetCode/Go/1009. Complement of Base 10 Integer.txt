func bitwiseComplement(n int) int {
    var sum = 0

	for counter := 0; true; counter++ {
		var bit = n & 1
		if bit == 0 {
			sum = sum + int(math.Pow(2, float64(counter)))
		}
		n = n >> 1
		if n <= 0 {
			break
		}

	}
	return sum
}
