func numberOfLines(widths []int, s string) []int {
    var lines = 0
	var sum = 0
	for i := 0; i < len(s); i++ {
		var currentpixel = widths[s[i]-97]
		sum = sum + currentpixel
		if sum > 100 {
			lines++
			sum = currentpixel
		}
	}
	var result = []int{lines + 1, sum}
	return result
}
